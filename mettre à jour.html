""" with open("loto_590_app/index.html", "w", encoding="utf-8") as f: f.write(index_html_content) # 2. scraper.py scraper_py_content = """import json import os import re import requests from bs4 import BeautifulSoup from datetime import datetime DATA_FILE = "data/tirages.json" LNB_URL = "https://www.lnb.bj" # Modifier selon la page exacte des resultats LNB def charger_historique(): if os.path.exists(DATA_FILE): try: with open(DATA_FILE, "r", encoding="utf-8") as f: return json.load(f) except Exception as e: print(f"Erreur de lecture JSON: {e}") return [] def sauvegarder_historique(data): os.makedirs(os.path.dirname(DATA_FILE), exist_ok=True) with open(DATA_FILE, "w", encoding="utf-8") as f: json.dump(data, f, ensure_ascii=False, indent=2) def extraire_tirage_lnb(): headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36" } try: response = requests.get(LNB_URL, headers=headers, timeout=15) if response.status_code == 200: soup = BeautifulSoup(response.text, "html.parser") # Recherche des elements contenant les boules du tirage # Adapter la classe CSS selon la structure effective du site LNB elements = soup.find_all("span", class_=re.compile(r"boule|ball|num", re.I)) numeros = [] for el in elements: val = el.text.strip() if val.isdigit() and 1 <= int(val) <= 90: numeros.append(val.zfill(2)) if len(numeros) == 5: break if len(numeros) == 5: return "-".join(numeros) except Exception as e: print(f"Erreur lors du scraping : {e}") return None def main(): historique = charger_historique() nouveau_tirage = extraire_tirage_lnb() if nouveau_tirage: # Verification anti-doublon (si le dernier tirage enregistre est different) dernier_enregistre = historique[0]["tirage"] if historique else None if nouveau_tirage != dernier_enregistre: date_str = datetime.now().strftime("%d/%m/%Y %H:%M") nouvel_element = { "tirage": nouveau_tirage, "date": date_str } historique.insert(0, nouvel_element) # Ajout en haut sauvegarder_historique(historique) print(f" Succes : Nouveau tirage ajoute -> {nouveau_tirage}") else: print(" Info : Le tirage est deja a jour.") else: print(" Attention : Aucun tirage valide n'a pu etre extrait lors de cette session.") if __name__ == "__main__": main() """ with open("loto_590_app/scraper.py", "w", encoding="utf-8") as f: f.write(scraper_py_content) #
